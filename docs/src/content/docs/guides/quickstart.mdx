## Step by Step:

### Define States

```tsx
const states = defineStates({
  Red: () => "means stop",
  Yellow: () => "means caution",
  Green: () => "means go",
})
```

### Define Transitions

```tsx
const transitions = {
  Red: { next: "Green" },
  Yellow: { next: "Red" },
  Green: { next: "Yellow" },
}
```

### Create Machine with Initial State

```tsx
const machine = createFactoryMachine(
  states, 
  transitions, 
  "Red"
)
```

### Use Machine (React)

```tsx
export const TrafficLight = () => {
  useMachine(trafficLight.machine);
  return (
   ...
  );
}
```

### `match` cases

```tsx
trafficLight.state.match({
  Red: () => "bg-red-500",
  Yellow: () => "bg-yellow-500",
  Green: () => "bg-green-500",
})
```

### Send machine events

```tsx
trafficLight.next()
```
